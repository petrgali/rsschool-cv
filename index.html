func Max(arr []int) int {
                swap := true
                count := 0
                for range arr {
                    count++
                }
                for swap {
                    swap = false
                    for k := 1; k < count; k++ {
                        for range arr {
                            if arr[k-1] > arr[k] {
                                arr[k], arr[k-1] = arr[k-1], arr[k]
                                swap = true
                            }
                        }
                    }
                }
                return arr[count-1]
            }
